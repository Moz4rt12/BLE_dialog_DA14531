103 __STATIC_INLINE void hw_otpc_init(void)
104 {
105  /*
106  * Reset OTP controller
107  */
108  SetWord32(OTPC_MODE_REG, HW_OTPC_MODE_STBY); // Set OTPC to standby
109  SetBits16(SYS_CTRL_REG, OTPC_RESET_REQ, 1); // Reset the OTPC
110  SetBits16(SYS_CTRL_REG, OTPC_RESET_REQ, 0); // Get the OTPC out of Reset
111 
112  /*
113  * Initialize the POR registers
114  */
115  SetWord32(OTPC_NWORDS_REG, OTPC_NWORDS_REG_RESET);
116  SetWord32(OTPC_TIM1_REG, OTPC_TIM1_REG_RESET);
117  SetWord32(OTPC_TIM2_REG, OTPC_TIM2_REG_RESET);
118 
119  /*
120  * Enable OTPC clock
121  */
122  SetBits16(CLK_AMBA_REG, OTP_ENABLE, 1);

128 }
