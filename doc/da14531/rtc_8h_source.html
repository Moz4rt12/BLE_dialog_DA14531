749 __STATIC_FORCEINLINE void rtc_reset(void)
750 {
751  uint8_t temp_reg;
752  temp_reg = GetWord32(RTC_INTERRUPT_MASK_REG);
753  /* Mask all interrupts to prevent any from being triggered due to block reset. */
754  SetWord32(RTC_INTERRUPT_DISABLE_REG, 0xFF);
755  SetBits32(CLK_RTCDIV_REG, RTC_RESET_REQ, 1);
756  SetBits32(CLK_RTCDIV_REG, RTC_RESET_REQ, 0);
757  /* Restore the interrupt mask. */
758  SetWord32(RTC_INTERRUPT_ENABLE_REG, ~temp_reg & 0x7F);
759 }
